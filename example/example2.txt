Step welcome
    Speak $name + "，您好，请问有什么可以帮您?"
    Speak "这是一条测试语句"
    Wait 5
    Branch "查询余额"
        billProc
    Branch "充值"
        rechargeProc
    Branch "查询机票"
        queryProc
    Branch "购买机票"
        buyProc
    Default
        defaultProc
    Exit


Step billProc
    Speak "您当前的余额是：" + $money
    Wait 100
    Branch "充值"
        rechargeProc
    Branch "查询机票"
        queryProc
    Branch "购买机票"
        buyProc
    Default
        defaultProc
    Exit

Step rechargeProc
    Speak "正在充值"
    Wait 100
    Branch "查询余额"
        billProc
    Branch "查询机票"
        queryProc
    Branch "购买机票"
        buyProc
    Default
        defaultProc
    Exit

Step queryProc
    Speak "正在查询机票"
    Wait 100
    Branch "购买机票"
        buyProc
    Default
        defaultProc
    Exit

Step buyProc
    Speak "正在购买机票"
    Wait 100
    Branch "查询机票"
        queryProc
    Branch "投诉"
        complainProc
    Default
        defaultProc
    Exit
